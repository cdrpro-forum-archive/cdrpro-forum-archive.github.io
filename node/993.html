Предыстория: установил наконец-то себе X5. Неприятный сюрприз мне преподнес мастер слой(МС). Если есть многостраничный документ и на него надо поставить МС, то он почему-то по умолчанию ставится поверх всех слоев (мне нужно позади), хотя на текущей странице все нормально. В инете толковых ответов не нашел, то ли никому это не надо, то ли я что-то не то делаю. Пришлось искать макрос, который ставит МС позади всех слоев. Нашел на форуме coreldraw.com

Sub Master2Bottom()
On Error Resume Next

     Dim d As Document
     Dim c As Integer, x As Integer, y As Integer, lc As Integer
     Dim z As String
     Set d = ActiveDocument
     c = ActiveDocument.Pages.Count
     z = ActivePage.ActiveLayer.Name
     For x = 1 To c
         d.Pages(x).Activate
         lc = ActivePage.Layers.Count
         ActivePage.Layers(z).Activate
         ActiveLayer.MoveBelow ActivePage.Layers(lc)
     Next x
      
End Sub
