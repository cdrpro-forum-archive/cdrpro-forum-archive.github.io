kuterma2, держи макрос для такого случая

Sub Out_Copy
Dim s As Shape
Dim s1 As Shape
     Set s = ActiveShape
     If Not s Is Nothing Then
         Set s1 = s.ParentGroup
         If Not s1 Is Nothing Then
             Set s = s.Duplicate
             s.OrderFrontOf s1
             s.CreateSelection
         End If
     End If
End Sub
