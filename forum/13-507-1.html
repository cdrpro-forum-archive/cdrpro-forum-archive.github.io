давно написал такой макрос, кому нужно - пользуйтесь
только подправьте для своих нужд строку c .Move

sub Clone_Down
Dim srS, srC As ShapeRange
     Set srS = ActiveSelectionRange
     With srS
         Set srC = .Duplicate
         .Move 0#, 0 - .SizeHeight ' копируем снизу от исходного  
         .OrderForwardOne
     End With
End Sub
